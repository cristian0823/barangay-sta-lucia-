${item.icon}
${item.name}
${item.description}
${item.available}/${item.quantity}
`; }).join(''); } async function requestEquipment(equipmentId) { const qty = prompt('How many do you need?', '1'); if (!qty || parseInt(qty) <= 0) return; const equipment=await getEquipment(); const item=equipment.find(e=> e.id === equipmentId); if (!item || item.available < parseInt(qty)) { showToast('Not enough items available.', 'error' ); return; } const purpose=prompt('What is the purpose of borrowing?'); if (!purpose) return; const today=new Date().toISOString().split('T')[0]; const returnDate=prompt('Return date (YYYY-MM-DD):', today); if (!returnDate) return; const result=await borrowEquipment(equipmentId, parseInt(qty), today, returnDate, purpose); if (result.success) { showToast('✅ ' + result.message, ' success'); await loadEquipment(); await updateStats(); } else showToast(result.message, 'error' ); } async function loadMyConcerns() { const concerns=await getMyConcerns(); const c=document.getElementById('myConcernsList'); if (!concerns.length) { c.innerHTML='

`; }).join(''); } document.getElementById('concernForm').addEventListener('submit', async function (e) { e.preventDefault(); const result = await submitConcern( document.getElementById('concernCategory').value, document.getElementById('concernTitle').value, document.getElementById('concernDescription').value, document.getElementById('concernAddress').value ); if (result.success) { showToast('✅ ' + result.message, 'success'); this.reset(); await loadMyConcerns(); await updateStats(); } else showToast(result.message, 'error'); }); async function loadMyBorrowings() { const borrowings = await getMyBorrowings(); const c = document.getElementById('myBorrowingsList'); if (!borrowings.length) { c.innerHTML = '
